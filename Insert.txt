import java.io.BufferedReader;
import java.io.FileReader;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class Insert {
    public static void main(String[] args) {
        try {
            // Establish a connection to the Derby database
            Connection conn = DriverManager.getConnection("jdbc:derby:/path_to_your_database");

            // Prepare a SQL statement for inserting data
            String insertSQL = "INSERT INTO lineitem VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)";
            PreparedStatement pstmt = conn.prepareStatement(insertSQL);

            // Read the .tbl file and insert data into the table
            BufferedReader reader = new BufferedReader(new FileReader("/path_to_your_data/lineitem.tbl"));
            String line;
            while ((line = reader.readLine()) != null) {
                // Split the line into individual values using the delimiter (|)
                String[] values = line.split("\\|");

                // Set parameters for the PreparedStatement
                for (int i = 0; i < values.length; i++) {
                    pstmt.setString(i + 1, values[i]);
                }

                // Execute the insert statement
                pstmt.executeUpdate();
            }

            // Close resources
            pstmt.close();
            conn.close();
            reader.close();
        } catch (SQLException | IOException e) {
            e.printStackTrace();
        }
    }
}
